    LIST	P=16F877A
    INCLUDE	"P16F877A.INC"
    SAYAC1   EQU 0X23
    SAYAC2   EQU 0X24
    REFERANS	EQU 0X25
    TEMP     EQU 0X26
    BIRLER   EQU 0X27
    ONLAR    EQU 0X28
    YUZLER   EQU 0X29
    OKUNAN_DEGER    EQU 0X30
    
    ORG	    0X00
    GOTO ANA
ANA
   BANKSEL  TRISB
   CLRF	    TRISB
   MOVLW    B'00000001'
   MOVWF    TRISA
   MOVLW    B'11000000'
   MOVWF    TRISD
   CLRF	    TRISC
   BANKSEL  ADCON1
   MOVLW    B'10001110'
   MOVWF    ADCON1
   BANKSEL  ADCON0
   MOVLW    B'11000001'
   MOVWF    ADCON0  
   BANKSEL  REFERANS
   MOVLW    H'0B'
   MOVWF    REFERANS  
   BANKSEL  REFERANS
   CLRF	PORTA
   CLRF	PORTB
   CLRF	PORTC
   CLRF	PORTD
   BANKSEL  ADRESL
   CLRF	ADRESL
   CALL	AYAR
   GOTO KONTROL

KONTROL
   CALL	    GECIKME
   BANKSEL  ADCON0
   BSF	    ADCON0,GO
   CALL	    GECIKME
   GOTO	    KIYASLA
   E2
   BANKSEL  PORTD
   BTFSC    PORTD,0
   CALL	    ARTTIR
   BTFSC    PORTD,1
   CALL	    AZALT
   GOTO	    KONTROL
   
KIYASLA
   BANKSEL  ADRESL
   RRF	    ADRESL
   MOVF	    ADRESL,W
   MOVWF    OKUNAN_DEGER
   CALL	    OKU
   GOTO	    KARAKTER
   BANKSEL  PORTC
   MOVWF    PORTC
   BANKSEL  REFERANS
   SUBWF    REFERANS,W
   BTFSS    STATUS,Z
   GOTO	    DUR
   GOTO	    CALIS

DUR
   BCF	    STATUS,C
   BANKSEL  PORTA
   BCF	    PORTA,3
   GOTO	    E2 
CALIS
   BCF	    STATUS,C
   BANKSEL  PORTA
   BSF	    PORTA,3
   GOTO	    E2   
ARTTIR
   BANKSEL  REFERANS
   MOVLW    D'1'
   ADDWF    REFERANS
   CALL	    GECIKME
   CALL	    GECIKME
   CALL	    KIYASLA
   RETURN   
AZALT
   BANKSEL  REFERANS
   MOVLW    D'1'
   SUBWF    REFERANS
   CALL	    GECIKME
   CALL	    GECIKME
   CALL	    KIYASLA
   RETURN 
   
AYAR   
   MOVLW    0X02
   CALL	    KOMUT_YAZ
   MOVLW    0X0F  
   CALL	    KOMUT_YAZ
   MOVLW    0X28
   CALL	    KOMUT_YAZ
   MOVLW    0X80
   CALL	    KOMUT_YAZ
   RETURN  
KOMUT_YAZ
    MOVWF TEMP
    SWAPF TEMP,W
    CALL KOMUT_GONDER
    MOVF TEMP,W
    CALL KOMUT_GONDER
    RETURN
KOMUT_GONDER
    ANDWF 0X0F
    BANKSEL PORTB
    MOVWF PORTB
    BCF PORTB,4
    BSF PORTB,5
    CALL GECIKME
    BCF PORTB,5
    RETURN
VERI_YAZ
    MOVWF TEMP
    SWAPF TEMP,W
    CALL VERI_GONDER
    MOVF TEMP,W
    CALL VERI_GONDER
    RETURN
VERI_GONDER
    ANDWF 0X0F
    MOVWF PORTB
    BSF PORTB,4
    BSF PORTB,5
    CALL GECIKME
    BCF PORTB,5
    RETURN 
KARAKTER
    CALL OKU
    MOVLW 0X01
    CALL KOMUT_YAZ

    MOVLW A'S'
    CALL VERI_YAZ
    MOVLW A'I'
    CALL VERI_YAZ
    MOVLW A'C'
    CALL VERI_YAZ
    MOVLW A'A'
    CALL VERI_YAZ
    MOVLW A'K'
    CALL VERI_YAZ
    MOVLW A'L'
    CALL VERI_YAZ
    MOVLW A'I'
    CALL VERI_YAZ
    MOVLW A'K'
    CALL VERI_YAZ
    MOVLW A':'
    CALL VERI_YAZ
    MOVF YUZLER,W
    CALL VERI_YAZ
    MOVF ONLAR,W
    CALL VERI_YAZ
    MOVF BIRLER,W
    CALL VERI_YAZ
    MOVLW A' '
    CALL VERI_YAZ
    MOVLW A'C'
    CALL VERI_YAZ
    CALL GECIKME2
    GOTO KONTROL
   
OKU
    CLRF YUZLER
    CLRF ONLAR
    CLRF BIRLER
    CLRF PORTB
    MOVF OKUNAN_DEGER,W
    MOVWF OKUNAN_DEGER   
YUZLER_ARTIR
    MOVLW D'100'
    SUBWF OKUNAN_DEGER,0
    BTFSS STATUS,C
    GOTO ONLAR_ARTIR
    MOVWF OKUNAN_DEGER
    INCF YUZLER,1
    GOTO YUZLER_ARTIR
ONLAR_ARTIR
    MOVLW D'10'
    SUBWF OKUNAN_DEGER,0
    BTFSS STATUS,C
    GOTO BIRLER_ARTIR
    MOVWF OKUNAN_DEGER
    INCF ONLAR,1
    GOTO ONLAR_ARTIR
BIRLER_ARTIR
    MOVF OKUNAN_DEGER,0
    MOVWF BIRLER
    MOVLW 0X30
    IORWF YUZLER,F
    IORWF ONLAR,F
    IORWF BIRLER,F
    RETURN   
   
GECIKME2
    MOVLW 0XFF	
    MOVWF SAYAC1	
    GOTO DON
GECIKME
    MOVLW 0X04
    MOVWF SAYAC1
DON
    MOVLW 0X8F
    MOVWF SAYAC2
DON2
    DECFSZ SAYAC2,1
    GOTO DON2
    DECFSZ SAYAC1,1
    GOTO DON
    RETURN
    
    
    END
